```python
def evaluate_data_quality(self, data):
    """评估数据质量"""
    score = 100
    issues = []
    
    # 检查 ROE
    if data['financial']['roe']:
        if data['financial']['roe'] < 5:
            score -= 30
            issues.append('ROE偏低')
        elif data['financial']['roe'] > 50:
            score -= 20
            issues.append('ROE偏高')
    
    # 检查市值
    if not data['financial']['total_mv']:
        score -= 25
        issues.append('市值数据缺失')
    elif data['financial']['total_mv'] < 1000000000:  # < 10亿
        score -= 20
        issues.append('市值数据可能错误')
    
    # 检查 PE/PB
    if not data['financial']['pe']:
        score -= 15
        issues.append('PE数据缺失')
    if not data['financial']['pb']:
        score -= 10
        issues.append('PB数据缺失')
    
    return {'score': max(0, score), 'issues': issues}
```
